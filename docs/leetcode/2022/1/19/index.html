impl Solution {
    pub fn contains_nearby_duplicate(mut nums: Vec<i32>, k: i32) -> bool {
        let k = k as usize;
        let mut map = std::collections::HashMap::new();
        for i in 0..nums.len() {
            if i > k {
                *map.entry(nums[i - k - 1]).or_insert(0) -= 1;
            }
            *map.entry(nums[i]).or_insert(0) += 1;
            if *map.entry(nums[i]).or_insert(0) > 1 {
                return true;
            }

        }
        return false;
    }
}
